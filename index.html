let streams = []; const FAVORITE_KEY = "jacq.favoriteFourStreams.v1"; const streamGrid = document.getElementById("streamGrid"); const selectedList = document.getElementById("selectedList"); const mainPlayer = document.getElementById("mainPlayer"); const mainTitle = document.getElementById("mainTitle"); const mainStatus = document.getElementById("mainStatus"); const columnSelect = document.getElementById("columnSelect"); let selectedIndexes = []; let currentMainIndex = 0; let mainMuted = false; // 🔥 LOAD JSON FIRST fetch("https://jacques1990.github.io/livestream/streams.json") .then(res => res.json()) .then(data => { streams = data; // ✅ ONLY run after streams loaded renderSelectedList(); showAllFeeds(); openMain(0, true); loadSavedPreset(); }); // ---------------------------- function withParams(baseUrl, mute = true){ const separator = baseUrl.includes("?") ? "&" : "?"; return `${baseUrl}${separator}autoplay=1&mute=${mute ? 1 : 0}`; } function openMain(index, unmute = true){ currentMainIndex = index; const s = streams[index]; mainTitle.textContent = s.name; mainStatus.textContent = `Focused on ${s.name}`; mainPlayer.src = withParams(s.url, !unmute ? true : false); mainMuted = !unmute; updateActiveCards(); } function updateActiveCards(){ document.querySelectorAll(".stream-card").forEach((card, idx) => { card.style.outline = idx === currentMainIndex ? "2px solid #e11d48" : "none"; }); } // ---------------------------- // UI BUILDING // ---------------------------- function buildCard(stream, index){ const card = document.createElement("div"); card.className = "stream-card"; card.innerHTML = `
${stream.name}
`; return card; } function renderGrid(indexes = streams.map((_, i) => i), columns = 2, mode = "all"){ streamGrid.innerHTML = ""; streamGrid.style.gridTemplateColumns = `repeat(${columns}, 1fr)`; indexes.forEach(i => { const card = buildCard(streams[i], i); if (mode === "four"){ card.querySelector(".stream-frame").style.height = "300px"; } streamGrid.appendChild(card); }); // attach events document.querySelectorAll(".focus-btn").forEach(btn => { btn.onclick = () => openMain(Number(btn.dataset.index), true); }); document.querySelectorAll(".solo-btn").forEach(btn => { btn.onclick = () => { openMain(Number(btn.dataset.index), true); requestFullscreen(); }; }); document.querySelectorAll(".pick-stream").forEach(box => { const idx = Number(box.dataset.index); box.checked = selectedIndexes.includes(idx); box.onchange = (e) => toggleSelection(idx, e.target.checked); }); updateActiveCards(); } // ---------------------------- // MODES // ---------------------------- function showAllFeeds(){ renderGrid(streams.map((_, i) => i), Number(columnSelect.value || 2), "all"); } function fourScreenMode(){ if (selectedIndexes.length !== 4){ alert("Select exactly 4 streams"); return; } renderGrid(selectedIndexes, 2, "four"); openMain(selectedIndexes[0], true); } // ---------------------------- // SELECTION // ---------------------------- function toggleSelection(index, checked){ if (checked){ if (selectedIndexes.length >= 4){ alert("Max 4 streams"); return; } selectedIndexes.push(index); } else { selectedIndexes = selectedIndexes.filter(v => v !== index); } renderSelectedList(); } function renderSelectedList(){ selectedList.innerHTML = ""; if (selectedIndexes.length === 0){ selectedList.innerHTML = "No streams selected"; return; } selectedIndexes.forEach(i => { const div = document.createElement("div"); div.className = "selected-chip"; div.innerText = streams[i]?.name || ""; selectedList.appendChild(div); }); } // ---------------------------- // PRESET // ---------------------------- function savePreset(){ if (selectedIndexes.length !== 4){ alert("Pick 4 streams first"); return; } localStorage.setItem(FAVORITE_KEY, JSON.stringify(selectedIndexes)); alert("Saved"); } function loadPreset(){ const raw = localStorage.getItem(FAVORITE_KEY); if (!raw) return; selectedIndexes = JSON.parse(raw); renderSelectedList(); fourScreenMode(); } function loadSavedPreset(){ const raw = localStorage.getItem(FAVORITE_KEY); if (!raw) return; selectedIndexes = JSON.parse(raw); renderSelectedList(); } // ---------------------------- // UTIL // ---------------------------- function requestFullscreen(){ const el = document.querySelector(".main-player-wrap"); if (el.requestFullscreen) el.requestFullscreen(); } function loadSplit(){ mainPlayer.src = "https://jacques1990.github.io/livestream/splitscreen.html"; } // ---------------------------- // BUTTONS // ---------------------------- document.getElementById("showAllBtn").onclick = showAllFeeds; document.getElementById("fourScreenBtn").onclick = fourScreenMode; document.getElementById("savePresetBtn").onclick = savePreset; document.getElementById("loadPresetBtn").onclick = loadPreset; document.getElementById("clearPresetBtn").onclick = () => { localStorage.removeItem(FAVORITE_KEY); selectedIndexes = []; renderSelectedList(); showAllFeeds(); }; // ---------------------------- // SHORTCUTS // ---------------------------- document.addEventListener("keydown", (e) => { if (!isNaN(e.key) && e.key !== "0"){ const idx = parseInt(e.key) - 1; if (idx < streams.length) openMain(idx, true); } if (e.key === "f") requestFullscreen(); if (e.key === "m") document.getElementById("muteToggle").click(); });
